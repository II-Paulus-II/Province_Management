			limit = {
				has_province_modifier = founding_city_in_progress

				has_province_modifier = founding_metropolis_in_progress

		}
		trigger_if = {
			limit = {
				has_province_modifier = urban_development_in_progress
			}
			NOT = {
				has_province_modifier = urban_development_in_progress
			}
		}
		trigger_if = {
			limit = {
				trade_goods = grain
			}
			NOT = {
				trade_goods = grain
			}
		}
		trigger_if = {
			limit = {
				trade_goods = fish
			}
			NOT = {
				trade_goods = fish
			}
		}
		trigger_if = {
			limit = {
				trade_goods = cattle
			}
			NOT = {
				trade_goods = cattle
			}
		}
		trigger_if = {
			limit = {
				trade_goods = vegetables
			}
			NOT = {
				trade_goods = vegetables
			}
		}
				
		trigger_if = {
			limit = {
				scope:player = {
					NOT = {
						can_pay_price = price_enact_policy
					}
